 from fastmcp import FastMCP

mcp = FastMCP("Servidor Demo")


@mcp.tool
def suma(a: int, b: int) -> int:
    """Suma dos números"""
    return a + b


@mcp.tool  # Colorin que es una etiqueta que indica que la función es una herramienta del servidor que se puede consumir
def saludar(nombre: str) -> str:
    return f"Hola {nombre}"


@mcp.tool
def crear_archivo(nombre_archivo: str, contenido: str) -> str:
    """Crea un archivo de texto con el contenido especificado."""
    try:
        with open(nombre_archivo, "w", encoding="utf-8") as archivo:
            archivo.write(contenido)
        return f"Archivo '{nombre_archivo}' creado correctamente"

    except Exception as e:
        return f"Error: {str(e)}"


if __name__ == "__main__":
    mcp.run()


    @mcp.tool
def crear_txt(nombre: str, apellido: str) -> str:
    """Crea un archivo de texto con el nombre '{nombre}_{apellido}.txt'.

    El contenido por defecto del archivo será:
    Nombre: {nombre}
    Apellido: {apellido}

    Devuelve un mensaje indicando éxito o el error ocurrido.
    """
    # Construir nombre de archivo 
    nombre_archivo = f"{nombre}_{apellido}.txt"
    contenido = f"Nombre: {nombre}\nApellido: {apellido}\n"
    try:
        with open(nombre_archivo, "w", encoding="utf-8") as ar:
            ar.write(contenido)
